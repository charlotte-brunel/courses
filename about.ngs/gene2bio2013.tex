   -R human_g1k_v37.fasta \
   -I Sample1_markdup.bam \
   -l INFO \
   -o recal_data.grp \
   -knownSites dbsnp_135.b37.vcf.gz \
   -L test.out.dir/capture500.bed \
   -cov ReadGroupCovariate \

   -cov CycleCovariate \
   -cov ContextCovariate
